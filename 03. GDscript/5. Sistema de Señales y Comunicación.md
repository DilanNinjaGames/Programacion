### 5.1. Definición de Señales Personalizadas (`signal`)

```gdscript
# Las señales son el sistema de eventos/observer de Godot
# Equivalen conceptualmente al patrón Observer o a los eventos de Qt
# Python no tiene equivalente nativo (se usan bibliotecas como PySignal o blinker)

extends Node

# ── SEÑAL SIN ARGUMENTOS ──────────────────────────────────────────────────
signal jugador_murio

# ── SEÑAL CON ARGUMENTOS TIPADOS ─────────────────────────────────────────
signal vida_cambiada(vida_actual: int, vida_maxima: int)
signal item_recogido(nombre_item: String, posicion: Vector2)
signal puntuacion_actualizada(nueva_puntuacion: int)

# ── SEÑALES HEREDADAS DE GODOT ────────────────────────────────────────────
# Los nodos de Godot ya vienen con señales predefinidas:
# Button:        pressed, toggled(button_pressed: bool)
# Area2D:        body_entered(body: Node2D), body_exited(body: Node2D)
# Timer:         timeout
# AnimationPlayer: animation_finished(anim_name: StringName)
# CharacterBody2D: (no tiene señales — usa código en _physics_process)
```

### 5.2. Conexión de Señales por Código y Mediante el Editor

```gdscript
extends Node

@onready var boton: Button = $Boton
@onready var temporizador: Timer = $Timer
@onready var area: Area2D = $Area2D

signal vida_cambiada(nueva_vida: int)

func _ready() -> void:

    # ── CONECTAR SEÑAL A MÉTODO DEL MISMO NODO ────────────────────────────
    # Forma: señal.connect(callable)
    boton.pressed.connect(_al_presionar_boton)

    # ── CONECTAR SEÑAL A MÉTODO CON ARGUMENTOS EXTRA (bind) ───────────────
    # bind() fija argumentos adicionales al callable
    boton.pressed.connect(_al_presionar_boton_con_id.bind(42, "especial"))

    # ── CONECTAR A LAMBDA (muy común en Godot 4) ──────────────────────────
    temporizador.timeout.connect(func():
        print("Tiempo agotado")
        temporizador.start()     # reiniciar el timer
    )

    # ── CONECTAR SEÑAL DE OTRO NODO A MÉTODO DE ESTE NODO ─────────────────
    area.body_entered.connect(_cuando_algo_entra)

    # ── SEÑAL PROPIA ──────────────────────────────────────────────────────
    vida_cambiada.connect(_al_cambiar_vida)

    # ── CONECTAR UNA VEZ (se desconecta automáticamente tras el primer emit) ─
    boton.pressed.connect(_al_presionar_boton, CONNECT_ONE_SHOT)

    # ── DESCONECTAR ───────────────────────────────────────────────────────
    boton.pressed.disconnect(_al_presionar_boton)

    # ── VERIFICAR SI ESTÁ CONECTADO ───────────────────────────────────────
    if boton.pressed.is_connected(_al_presionar_boton):
        print("Está conectado")

func _al_presionar_boton() -> void:
    print("¡Botón presionado!")

func _al_presionar_boton_con_id(id: int, tipo: String) -> void:
    print("Botón ", id, " tipo: ", tipo)

func _cuando_algo_entra(cuerpo: Node2D) -> void:
    print("Entró al área: ", cuerpo.name)

func _al_cambiar_vida(nueva_vida: int) -> void:
    print("Nueva vida: ", nueva_vida)
```

**Conectar desde el Editor (Inspector):**

```
1. Seleccionar el nodo emisor en la escena
2. Panel "Nodo" (pestaña junto a Inspector) → sección "Señales"
3. Doble click en la señal deseada
4. Seleccionar el nodo receptor y el método destino
5. "Conectar" → Godot genera el método con prefijo _on_NodoEmisor_nombre_señal
   El editor agrega el ícono de conexión en el margen del código
```

### 5.3. Emisión de Señales con Argumentos

```gdscript
extends CharacterBody2D

class_name Jugador

signal vida_cambiada(vida_actual: int, vida_maxima: int)
signal jugador_murio
signal item_recogido(nombre: String, valor: int)

var vida: int = 100
var vida_maxima: int = 100

func recibir_danio(cantidad: int) -> void:
    vida = clamp(vida - cantidad, 0, vida_maxima)  # clamp = min(max(x, min_val), max_val)

    # Emitir señal con argumentos — todos los conectados son notificados
    vida_cambiada.emit(vida, vida_maxima)

    if vida <= 0:
        jugador_murio.emit()    # sin argumentos
        queue_free()

func recoger_item(nombre: String, valor: int) -> void:
    # Emitir señal con múltiples argumentos
    item_recogido.emit(nombre, valor)

# ── EJEMPLO: HUD que escucha las señales del Jugador ─────────────────────
# Archivo: hud.gd
extends CanvasLayer

@onready var barra_vida: ProgressBar = $BarraVida
@onready var label_vida: Label = $LabelVida

func _ready() -> void:
    # Obtener el jugador y conectar sus señales
    var jugador = get_tree().get_first_node_in_group("jugadores")
    if jugador:
        jugador.vida_cambiada.connect(_actualizar_vida)
        jugador.jugador_murio.connect(_mostrar_pantalla_muerte)

func _actualizar_vida(vida_actual: int, vida_maxima: int) -> void:
    barra_vida.value = float(vida_actual) / float(vida_maxima) * 100.0
    label_vida.text = "%d / %d" % [vida_actual, vida_maxima]

func _mostrar_pantalla_muerte() -> void:
    $PantallaGameOver.visible = true
```
