```gdscript
# ══════════════════════════════════════════════════════
# ARRAYS (equivalente a list de Python)
# ══════════════════════════════════════════════════════

var items: Array = []                        # vacío
var numeros: Array[int] = [1, 2, 3, 4, 5]  # tipado (Godot 4)

# Métodos principales — comparación con Python:
numeros.append(6)               # Python: .append(6)    ← igual
numeros.push_back(7)            # Python: no existe     — alias de append
numeros.push_front(0)           # Python: .insert(0, x)
numeros.pop_back()              # Python: .pop()
numeros.pop_front()             # Python: .pop(0)       — O(n)!
numeros.insert(2, 99)           # Python: .insert(2, 99) ← igual
numeros.remove_at(2)            # Python: .pop(2) o del lista[2]
numeros.erase(99)               # Python: .remove(99) — elimina primer valor
numeros.clear()                 # Python: .clear() o = []

print(numeros.size())           # Python: len(numeros)
print(numeros.is_empty())       # Python: len(numeros) == 0
print(numeros.has(5))           # Python: 5 in numeros
print(numeros.count(2))         # Python: numeros.count(2)   ← igual

numeros.sort()                  # Python: numeros.sort()      ← igual
numeros.shuffle()               # Python: random.shuffle(numeros)
numeros.reverse()               # Python: numeros.reverse()   ← igual

# Slicing: NO existe la sintaxis lista[1:3] de Python
# Usar slice() o loops manuales
var sub = numeros.slice(1, 3)   # equivale a numeros[1:3]

# Comprensión de arrays (GDScript NO tiene list comprehensions)
# Python: cuadrados = [x**2 for x in range(5)]
# GDScript: loop manual
var cuadrados: Array[int] = []
for x in range(5):
    cuadrados.append(x * x)

# Tipos de Array especializados (mucho más eficientes — sin equivalente Python)
var bytes: PackedByteArray    = PackedByteArray([0, 255, 128])
var floats: PackedFloat32Array = PackedFloat32Array([1.0, 2.0, 3.0])
var strings: PackedStringArray = PackedStringArray(["a", "b", "c"])
var vec2s: PackedVector2Array  = PackedVector2Array()
# Usar PackedArrays para datos de rendimiento crítico (ej: vértices de mesh)

# ══════════════════════════════════════════════════════
# DICCIONARIOS (equivalente a dict de Python)
# ══════════════════════════════════════════════════════

var config: Dictionary = {
    "velocidad": 300.0,
    "salud": 100,
    "nombre": "Héroe"
}

# Acceso — dos formas (igual que Python con [] y atributo):
print(config["velocidad"])           # Python: config["velocidad"]
print(config.get("mana", 0))         # Python: config.get("mana", 0)   ← igual

# Modificar
config["velocidad"] = 350.0          # Python: config["velocidad"] = 350.0
config["nuevo_key"] = true           # Python: config["nuevo_key"] = True

# Verificar existencia
if config.has("salud"):              # Python: "salud" in config
    print("tiene salud")

# Métodos
print(config.keys())                 # Python: config.keys()   ← igual
print(config.values())               # Python: config.values() ← igual
config.erase("nombre")               # Python: del config["nombre"]
config.clear()                       # Python: config.clear()  ← igual
print(config.is_empty())             # Python: len(config) == 0
print(config.size())                 # Python: len(config)
```
