
```gdscript
extends Node

# GDScript usa sintaxis de propiedad con set y get inline
# Python: @property y @x.setter — GDScript: var x: tipo: get = ..., set = ...

# ── SETTER Y GETTER COMPLETOS ─────────────────────────────────────────────
var vida: int = 100:
    get:                         # se ejecuta al LEER la variable
        return vida
    set(nuevo_valor):            # se ejecuta al ESCRIBIR la variable
        vida = clamp(nuevo_valor, 0, vida_maxima)
        vida_cambiada.emit(vida)   # efecto secundario automático
        if vida == 0:
            jugador_murio.emit()

# ── SOLO GETTER (read-only desde fuera) ──────────────────────────────────
var vida_maxima: int = 100
var porcentaje_vida: float:
    get:
        return float(vida) / float(vida_maxima)
    # sin set → la variable es de solo lectura

# ── SOLO SETTER (write-only — raro pero posible) ─────────────────────────
var _color_interno: Color = Color.WHITE
var color: Color:
    set(nuevo_color):
        _color_interno = nuevo_color
        $Sprite2D.modulate = nuevo_color    # aplicar cambio visual automáticamente

# ── COMBINANDO @export CON SETTER ─────────────────────────────────────────
# Muy útil: el diseñador cambia el valor en el Inspector y el setter reacciona
@export var velocidad: float = 300.0:
    set(valor):
        velocidad = max(0.0, valor)          # no permitir velocidad negativa
        _actualizar_animacion_velocidad()    # reaccionar al cambio inmediatamente
    get:
        return velocidad

signal vida_cambiada(nueva_vida: int)
signal jugador_murio

func _ready() -> void:
    # Probar el setter
    vida = 150    # el setter lo recorta a 100 (clamp al máximo)
    print(vida)   # 100

    vida = -10    # el setter lo recorta a 0 y emite jugador_murio
    print(porcentaje_vida)  # 0.0

func _actualizar_animacion_velocidad() -> void:
    pass  # lógica de animación según velocidad
```
