Estas funciones son **callbacks automáticos** que Godot llama en momentos específicos. El prefijo `_` indica que son métodos virtuales sobreescritos.

```gdscript
extends Node

# ── _init(): constructor ───────────────────────────────────────────────────
# Se llama cuando el objeto se crea en MEMORIA (antes de entrar al árbol)
# Python equivalente: __init__(self)
# ⚠️ En Godot: raramente se usa. Preferir _ready() para lógica de inicio.
func _init() -> void:
    print("Objeto creado en memoria")
    # No usar get_node() aquí — el nodo aún no está en el árbol de escena

# ── _ready(): inicio post-árbol ───────────────────────────────────────────
# Se llama cuando el nodo Y TODOS SUS HIJOS han entrado al árbol de escena
# Es el lugar correcto para inicialización de lógica de juego
# Equivale conceptualmente a __post_init__ o a un setUp()
func _ready() -> void:
    print("Árbol listo. Nodo: ", name)
    # Aquí SÍ se puede usar get_node(), $NodoHijo, etc.
    $Sprite2D.visible = true

# ── _process(delta): actualización por frame ──────────────────────────────
# Se llama cada frame del motor (60fps → ~60 veces/segundo)
# delta: tiempo en SEGUNDOS desde el frame anterior (generalmente ~0.016)
# Usar para: lógica de juego, input, animaciones suaves
func _process(delta: float) -> void:
    # Multiplicar por delta para que el movimiento sea independiente del FPS
    position.x += 100.0 * delta     # mueve 100 unidades POR SEGUNDO (no por frame)

# ── _physics_process(delta): actualización física ─────────────────────────
# Se llama a frecuencia FIJA (por defecto 60 veces/segundo, configurable)
# Independiente del FPS de renderizado
# Usar para: movimiento de personajes, física, detección de colisiones
func _physics_process(delta: float) -> void:
    # Para CharacterBody2D: mover con el motor de física
    velocity.y += 980.0 * delta     # gravedad
    move_and_slide()                 # método de CharacterBody2D — aplica velocity

# ── _input(event): input global ───────────────────────────────────────────
# Recibe TODOS los eventos de input (teclado, mouse, gamepad)
func _input(event: InputEvent) -> void:
    if event is InputEventKey and event.pressed:
        if event.keycode == KEY_ESCAPE:
            get_tree().quit()

# ── _unhandled_input(event): input no consumido ───────────────────────────
# Solo recibe inputs que NO fueron consumidos por la UI u otros nodos
func _unhandled_input(event: InputEvent) -> void:
    if event.is_action_pressed("saltar"):
        saltar()

# ── _notification(what): notificaciones del motor ─────────────────────────
# Godot envía notificaciones en eventos del ciclo de vida del nodo
func _notification(what: int) -> void:
    match what:
        NOTIFICATION_WM_CLOSE_REQUEST:   # ventana cerrándose
            guardar_datos()
        NOTIFICATION_PAUSED:             # juego pausado
            print("Juego pausado")
```
