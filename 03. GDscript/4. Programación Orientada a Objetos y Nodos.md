### 4.1. Clases, Herencia y `class_name`

```gdscript
# ── ARCHIVO COMO CLASE (lo más común en Godot) ───────────────────────────
# Archivo: enemigo.gd
extends CharacterBody2D

class_name Enemigo    # registra esta clase globalmente por nombre
                      # otros scripts pueden hacer: var e = Enemigo.new()
                      # sin necesidad de load() o preload()

var vida: int = 50
var velocidad: float = 100.0

func recibir_danio(cantidad: int) -> void:
    vida -= cantidad
    if vida <= 0:
        morir()

func morir() -> void:
    queue_free()      # equivale a destruir el nodo y eliminarlo del árbol

# ── CLASES INTERNAS (nested classes — como clases internas en Python) ──────
# Declaradas dentro de un script pero no son el script principal
class DatosDeNivel:
    var nombre: String
    var puntuacion_maxima: int = 0

    func _init(p_nombre: String) -> void:
        nombre = p_nombre

# Usar la clase interna:
var nivel_1 = DatosDeNivel.new("Bosque Oscuro")

# ── HERENCIA EN ARCHIVO SEPARADO ──────────────────────────────────────────
# Archivo: enemigo_volador.gd
extends Enemigo      # hereda del script enemigo.gd
                     # Python equiv: class EnemigVolador(Enemigo):

class_name EnemigoVolador

var altura_vuelo: float = 200.0

func _ready() -> void:
    super._ready()              # llamar al _ready() del padre
    # Python: super()._ready()  — en GDScript: super.metodo()

func recibir_danio(cantidad: int) -> void:
    # Los voladores reciben la mitad del daño
    super.recibir_danio(cantidad / 2)    # llamar al método del padre
```

### 4.2. Instanciación de Escenas y Scripts

```gdscript
# ── PRELOAD vs LOAD ────────────────────────────────────────────────────────
# preload(): carga el recurso en TIEMPO DE COMPILACIÓN
#            rápido en runtime, falla en compilación si la ruta es inválida
# load():    carga el recurso en TIEMPO DE EJECUCIÓN
#            flexible, puede fallar en runtime si no existe la ruta

# Python no tiene equivalente directo — similar a importar módulos

# Cargar una escena
const ESCENA_BALA = preload("res://proyectiles/bala.tscn")  # en tiempo de compilación
var escena_enemigo = load("res://enemigos/goblin.tscn")      # en tiempo de ejecución

# Cargar un script
const ScriptEnemigo = preload("res://scripts/enemigo.gd")

# ── INSTANCIAR UNA ESCENA ─────────────────────────────────────────────────
func _ready() -> void:

    # instantiate() crea una INSTANCIA de la escena (objeto en memoria)
    var nueva_bala: Node = ESCENA_BALA.instantiate()

    # El nodo instanciado debe agregarse al árbol para ser procesado
    add_child(nueva_bala)               # agregar como hijo de este nodo
    get_tree().root.add_child(nueva_bala) # agregar a la raíz de la escena
    get_parent().add_child(nueva_bala)  # agregar como hermano

    # Configurar propiedades ANTES o DESPUÉS de add_child
    nueva_bala.position = Vector2(100, 200)
    nueva_bala.velocidad = 500.0         # si la bala tiene esa variable exportada

# ── INSTANCIAR UN SCRIPT (sin escena .tscn) ────────────────────────────────
# Para clases puras sin representación visual
var calculadora = ScriptEnemigo.new()   # Python: calculadora = Enemigo()
# o si tiene class_name:
var enemigo = Enemigo.new()

# ── ELIMINAR NODOS ────────────────────────────────────────────────────────
nodo.queue_free()    # elimina AL FINAL del frame actual (seguro — más común)
nodo.free()          # elimina INMEDIATAMENTE (puede causar errores si hay refs)
```

### 4.3. Acceso a Nodos del Árbol y `@onready`

```gdscript
extends Node2D

# ── $NodoHijo: azúcar sintáctico para get_node() ─────────────────────────
# Equivale a buscar un hijo por nombre en el árbol de escena
# Solo válido DESPUÉS de _ready() (el árbol debe estar construido)

func _ready() -> void:
    # Formas equivalentes de acceder al mismo nodo:
    var sprite_1 = get_node("Sprite2D")         # forma explícita con String
    var sprite_2 = $Sprite2D                    # azúcar — más corto
    var sprite_3 = get_node(^"Sprite2D")        # NodePath explícito (mismo que $)

    # Acceso a nodos anidados (rutas con /)
    var label = $UI/PanelContenedor/LabelVida   # navega la jerarquía

    # Acceso al padre
    var padre = get_parent()

    # Acceso a la raíz de la escena
    var raiz = get_tree().current_scene

    # Acceso a un nodo por ruta absoluta desde la raíz
    var hud = get_node("/root/EscenaPrincipal/HUD")

# ── @onready: resolver referencias EN _ready() automáticamente ────────────
# Problema sin @onready: acceder a $NodoHijo en la declaración de variable falla
# porque el árbol no está listo aún en ese momento.

# ❌ INCORRECTO — falla porque el árbol no está listo durante la inicialización
# var sprite: Sprite2D = $Sprite2D    # Error: nodo no existe todavía

# ✅ @onready — asigna la variable CUANDO _ready() se ejecuta automáticamente
@onready var sprite: Sprite2D        = $Sprite2D
@onready var label_vida: Label       = $UI/LabelVida
@onready var animaciones: AnimationPlayer = $AnimationPlayer
@onready var colision: CollisionShape2D = $CollisionShape2D

# Equivale exactamente a escribir esto en _ready():
# func _ready() -> void:
#     sprite       = $Sprite2D
#     label_vida   = $UI/LabelVida
#     animaciones  = $AnimationPlayer

func _ready() -> void:
    # Aquí @onready ya asignó las variables — pueden usarse directamente
    label_vida.text = "Vida: 100"
    animaciones.play("idle")

# ── BUSCAR NODOS DINÁMICAMENTE ────────────────────────────────────────────
# Útil cuando no sabes la posición exacta en el árbol

func _ready() -> void:
    # Buscar nodo por nombre en todos los descendientes (lento — evitar en _process)
    var cualquier_label = find_child("*Label*", true, false)  # wildcard

    # Buscar nodos de un tipo específico en el árbol completo
    var todos_los_enemigos = get_tree().get_nodes_in_group("enemigos")
    for enemigo in todos_los_enemigos:
        enemigo.alertar()

    # Agregar este nodo a un grupo (para encontrarlo desde otros scripts)
    add_to_group("jugadores")
```
