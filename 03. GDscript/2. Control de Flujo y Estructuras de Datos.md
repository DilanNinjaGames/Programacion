### 2.1. Condicionales (`if`, `elif`, `else`) y Operador `match`

```gdscript
# ── IF / ELIF / ELSE — idéntico a Python ──────────────────────────────────
var puntos: int = 85

if puntos >= 90:
    print("Excelente")
elif puntos >= 70:
    print("Bien")
else:
    print("Sigue intentando")

# ── MATCH — el switch/case de GDScript (más potente que Python 3.10 match) ─
# Python: match valor: case x: ...
# GDScript: match valor: x: ...  ← sin la palabra 'case'

var estado: String = "corriendo"

match estado:
    "idle":
        print("En reposo")
    "corriendo":
        print("Corriendo")
    "saltando":
        print("En el aire")
    _:                          # _ es el default (como case _ en Python)
        print("Estado desconocido")

# MATCH con múltiples valores en un caso (como case 1 | 2 en Python)
var tecla: int = 2

match tecla:
    1, 2, 3:                    # si tecla es 1 O 2 O 3
        print("Número del 1 al 3")
    4, 5:
        print("4 o 5")
    _:
        print("Otro")

# MATCH con rangos usando when (guard clause — GDScript 4)
var vida: int = 65

match true:                     # truco: matchear true para usar condiciones
    vida > 80:
        print("Saludable")
    vida > 50:
        print("Herido")
    _:
        print("Crítico")

# MATCH con tipos de objeto
func procesar(valor) -> void:
    match typeof(valor):
        TYPE_INT:
            print("Entero: ", valor)
        TYPE_STRING:
            print("Texto: ", valor)
        TYPE_ARRAY:
            print("Array de ", len(valor), " elementos")
```

### 2.2. Bucles (`for` en Rangos/Arrays, `while`)

```gdscript
# ── FOR con range() — idéntico a Python ───────────────────────────────────
for i in range(5):              # 0, 1, 2, 3, 4
    print(i)

for i in range(2, 10, 2):      # 2, 4, 6, 8
    print(i)

for i in range(10, 0, -1):     # 10, 9, 8... 1 (decreciente)
    print(i)

# ── FOR sobre Array (como Python for x in lista) ──────────────────────────
var enemigos: Array[String] = ["Slime", "Goblin", "Dragon"]

for enemigo in enemigos:
    print("Encontré: ", enemigo)

# FOR con índice (no hay enumerate() — usar range + índice)
for i in range(enemigos.size()):    # .size() en vez de len()
    print(i, ": ", enemigos[i])

# ── FOR sobre Dictionary (como Python for k, v in d.items()) ──────────────
var stats: Dictionary = {"ataque": 10, "defensa": 5, "velocidad": 8}

for clave in stats:                  # itera sobre las claves
    print(clave, " = ", stats[clave])

# ── WHILE — idéntico a Python ─────────────────────────────────────────────
var contador: int = 0

while contador < 5:
    print(contador)
    contador += 1

# ── BREAK y CONTINUE — idénticos a Python ─────────────────────────────────
for i in range(10):
    if i == 3:
        continue                # saltar esta iteración
    if i == 7:
        break                   # salir del bucle
    print(i)                    # imprime 0,1,2,4,5,6
```
