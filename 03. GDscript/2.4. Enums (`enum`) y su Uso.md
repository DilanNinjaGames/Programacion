# Python no tiene enum nativo en la misma forma — usa el módulo enum
# GDScript tiene enum como ciudadano de primera clase

# ── ENUM SIMPLE (las constantes tienen valores 0, 1, 2... automáticamente) ─
enum Estado {
    IDLE,          # = 0
    CORRIENDO,     # = 1
    SALTANDO,      # = 2
    MURIENDO       # = 3
}

# ── ENUM CON VALORES EXPLÍCITOS ───────────────────────────────────────────
enum Dificultad {
    FACIL   = 1,
    NORMAL  = 2,
    DIFICIL = 5,
    EXTREMO = 10
}

# ── USO ──────────────────────────────────────────────────────────────────
var estado_actual: Estado = Estado.IDLE

func cambiar_estado(nuevo: Estado) -> void:
    estado_actual = nuevo
    match estado_actual:
        Estado.IDLE:
            print("Quieto")
        Estado.CORRIENDO:
            print("Corriendo")

# Acceder al valor numérico
print(Estado.CORRIENDO)              # imprime 1

# Verificar tipo
if estado_actual == Estado.IDLE:
    print("estoy idle")

# ENUM DENTRO DE UNA CLASE (tiene namespace propio)
class_name Jugador
extends CharacterBody2D

enum PowerUp {
    VELOCIDAD,
    ESCUDO,
    DOBLE_SALTO
}

var powerup_activo: PowerUp = PowerUp.VELOCIDAD
