```gdscript
# GDScript 4 tiene lambdas (funciones anónimas) con la palabra clave func
# Python: lambda x: x * 2
# GDScript: func(x): return x * 2

# ── LAMBDA BÁSICA ─────────────────────────────────────────────────────────
var duplicar = func(x: int) -> int:
    return x * 2

print(duplicar.call(5))     # → 10
# .call() es necesario para invocar un Callable almacenado en variable

# ── LAMBDA MULTI-LÍNEA ────────────────────────────────────────────────────
var procesar = func(valor: float) -> float:
    var resultado = valor * 2.0
    resultado += 10.0
    return resultado

print(procesar.call(5.0))   # → 20.0

# ── CALLABLE: referencia a función existente ──────────────────────────────
# Equivale a pasar una función como argumento en Python
func al_completar(mensaje: String) -> void:
    print("Completado: ", mensaje)

var mi_callable: Callable = al_completar
mi_callable.call("nivel 1")   # → "Completado: nivel 1"

# ── USO PRINCIPAL: conectar señales y callbacks ──────────────────────────
# Conectar señal con lambda (muy común en Godot 4)
$Boton.pressed.connect(func():
    print("Botón presionado via lambda")
    cambiar_escena()
)

# Usar con métodos funcionales de Array
var numeros: Array[int] = [1, 2, 3, 4, 5, 6]

# filter() — equivale a filter() de Python
var pares = numeros.filter(func(n: int) -> bool: return n % 2 == 0)
print(pares)   # [2, 4, 6]

# map() — equivale a map() de Python
var cuadrados = numeros.map(func(n: int) -> int: return n * n)
print(cuadrados)   # [1, 4, 9, 16, 25, 36]

# reduce() — equivale a functools.reduce() de Python
var suma = numeros.reduce(func(acc: int, n: int) -> int: return acc + n, 0)
print(suma)   # 21

# any() / all() — equivale a Python
var hay_negativo = numeros.any(func(n: int) -> bool: return n < 0)
var todos_positivos = numeros.all(func(n: int) -> bool: return n > 0)
```
