### 3.1. Declaración de Funciones (`func`) y Retorno de Tipos

```gdscript
# ── FUNCIÓN BÁSICA ─────────────────────────────────────────────────────────
# Python: def saludar(nombre: str) -> str:
func saludar(nombre: String) -> String:
    return "Hola, " + nombre

# ── SIN RETORNO (equivale a -> None en Python) ────────────────────────────
func reiniciar() -> void:
    vida = 100
    posicion = Vector2.ZERO

# ── PARÁMETROS CON VALOR POR DEFECTO ──────────────────────────────────────
# Python: def mover(dx: float, dy: float = 0.0):
func mover(dx: float, dy: float = 0.0) -> void:
    position.x += dx
    position.y += dy

# ── PARÁMETROS SIN TIPO (dinámicos — evitar en código de producción) ──────
func procesar_dato(dato) -> void:
    print(dato)

# ── RETORNO MÚLTIPLE (igual que Python — retorna Array) ───────────────────
func obtener_rango() -> Array[int]:
    return [vida_min, vida_max]     # no hay tuplas — usar Array

# ── FUNCIONES ESTÁTICAS ───────────────────────────────────────────────────
# Pueden llamarse sin instancia: MiScript.calcular_distancia(a, b)
static func calcular_distancia(a: Vector2, b: Vector2) -> float:
    return a.distance_to(b)        # método nativo de Vector2

# ── VARIADIC (sin equivalente directo a *args — usar Array) ───────────────
func sumar_todo(numeros: Array[float]) -> float:
    var total: float = 0.0
    for n in numeros:
        total += n
    return total
```
