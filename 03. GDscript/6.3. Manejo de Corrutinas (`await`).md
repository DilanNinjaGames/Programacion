
```gdscript
# await en GDScript es similar al async/await de Python
# pero sin necesidad de declarar la función como 'async' explícitamente
# En GDScript 4: cualquier función que use 'await' se vuelve una corrutina

extends Node

# ── AWAIT CON SEÑALES ─────────────────────────────────────────────────────
# Uso más común: pausar la ejecución hasta que una señal se emita

func secuencia_de_juego() -> void:
    print("Iniciando secuencia...")

    # Esperar a que el Timer termine (pausa esta función, NO el juego)
    await $Timer.timeout          # el motor sigue corriendo mientras espera
    print("Timer completado")

    # Esperar a que una animación termine
    $AnimationPlayer.play("entrada")
    await $AnimationPlayer.animation_finished
    print("Animación de entrada completada")

    # Esperar la señal de una escena cargada
    var escena = ResourceLoader.load_threaded_get("res://nivel2.tscn")
    await ResourceLoader.load_threaded_request("res://nivel2.tscn")
    print("Escena cargada")

# ── AWAIT CON FRAMES ──────────────────────────────────────────────────────
func esperar_frames() -> void:
    print("Frame actual")
    await get_tree().process_frame    # espera exactamente 1 frame
    print("Frame siguiente")

    # Esperar N frames
    for i in 5:
        await get_tree().process_frame
    print("5 frames después")

# ── CREATE_TIMER: pausa por tiempo sin usar nodo Timer ────────────────────
func con_temporizador() -> void:
    print("Inicio")
    await get_tree().create_timer(2.0).timeout   # espera 2 segundos
    print("Pasaron 2 segundos")
    await get_tree().create_timer(0.5).timeout   # espera 0.5 segundos
    print("Pasaron 0.5 segundos más")

# ── CORRUTINA QUE RETORNA VALOR ───────────────────────────────────────────
func cargar_datos_async() -> Dictionary:
    await get_tree().create_timer(1.0).timeout   # simular carga
    return {"nivel": 1, "puntuacion": 0}

func _ready() -> void:
    # Llamar a la corrutina y esperar su resultado
    var datos = await cargar_datos_async()
    print("Datos cargados: ", datos)

# ── AWAIT CON SEÑAL PROPIA ────────────────────────────────────────────────
signal operacion_completada(resultado: String)

func operacion_larga() -> void:
    await get_tree().create_timer(3.0).timeout
    operacion_completada.emit("éxito")

func iniciar_y_esperar() -> void:
    operacion_larga()                          # iniciar sin bloquear
    var resultado = await operacion_completada # esperar el resultado
    print("Resultado: ", resultado)

# ── DIFERENCIA CLAVE CON PYTHON ───────────────────────────────────────────
# Python async/await: la función DEBE marcarse con 'async def'
# GDScript await: cualquier función que use await se convierte en corrutina
#                 automáticamente — sin decorador ni palabra clave extra
# Python asyncio corre en un event loop propio
# GDScript await está integrado en el loop del motor (SceneTree)
```
