
```gdscript
extends CharacterBody2D

# @export hace la variable visible y editable en el Inspector de Godot
# Equivale a una configuración que el diseñador puede cambiar sin tocar el código
# Python no tiene equivalente — es específico del flujo de trabajo de Godot

# ── TIPOS BÁSICOS ─────────────────────────────────────────────────────────
@export var velocidad: float = 300.0
@export var vida_maxima: int = 100
@export var nombre_personaje: String = "Héroe"
@export var activo: bool = true
@export var color_base: Color = Color.WHITE

# ── RANGOS (muestra un slider en el Inspector) ────────────────────────────
@export_range(0.0, 1000.0, 10.0) var velocidad_con_rango: float = 300.0
@export_range(1, 10) var nivel: int = 1

# ── RECURSOS Y NODOS ──────────────────────────────────────────────────────
@export var icono: Texture2D                        # selector de imagen en Inspector
@export var escena_proyectil: PackedScene           # arrastra una .tscn desde FileSystem
@export var audio_disparo: AudioStream              # selector de audio

# ── ENUM COMO DESPLEGABLE ─────────────────────────────────────────────────
enum TipoEnemigo { NORMAL, ELITE, JEFE }
@export var tipo: TipoEnemigo = TipoEnemigo.NORMAL  # muestra dropdown en Inspector

# ── ARRAYS EXPORTADOS ─────────────────────────────────────────────────────
@export var puntos_de_patrulla: Array[Vector2] = []  # array editable en Inspector
@export var items_iniciales: Array[String] = []

# ── GRUPOS DE EXPORT (Godot 4.1+) ────────────────────────────────────────
@export_group("Combate")             # crea sección visual en el Inspector
@export var danio_base: int = 10
@export var cadencia_disparo: float = 0.3
@export var alcance: float = 500.0

@export_group("Movimiento")
@export var velocidad_caminar: float = 200.0
@export var velocidad_correr: float = 400.0

@export_group("")                    # cierra el grupo (vuelve al default)
@export var debug_mode: bool = false

# ── SUBGRUPOS ─────────────────────────────────────────────────────────────
@export_subgroup("Resistencias")
@export var resistencia_fuego: float = 0.0
@export var resistencia_hielo: float = 0.0
```
