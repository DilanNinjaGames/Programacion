/**
 * Representa una cuenta bancaria con operaciones básicas.
 *
 * <p>Esta clase implementa las operaciones de depósito y retiro
 * con validación de saldo. Uso básico:</p>
 *
 * <pre>{@code
 * CuentaBancaria cuenta = new CuentaBancaria("Ana", 1000.0);
 * cuenta.depositar(500.0);
 * cuenta.retirar(200.0);
 * }</pre>
 *
 * @author  Tu Nombre
 * @version 1.0
 * @since   Java 17
 */
public class CuentaBancaria {

    /** Nombre del titular de la cuenta. */
    private String titular;

    /** Saldo actual en la cuenta (nunca negativo). */
    private double saldo;

    /**
     * Crea una nueva cuenta bancaria con saldo inicial.
     *
     * @param titular nombre del propietario de la cuenta
     * @param saldoInicial monto inicial — debe ser mayor o igual a cero
     * @throws IllegalArgumentException si {@code saldoInicial} es negativo
     */
    public CuentaBancaria(String titular, double saldoInicial) {
        if (saldoInicial < 0) {
            throw new IllegalArgumentException("El saldo inicial no puede ser negativo");
        }
        this.titular = titular;
        this.saldo   = saldoInicial;
    }

    /**
     * Deposita un monto en la cuenta.
     *
     * @param  monto cantidad a depositar — debe ser positiva
     * @return el nuevo saldo tras el depósito
     * @throws IllegalArgumentException si {@code monto} es menor o igual a cero
     * @see    #retirar(double)
     */
    public double depositar(double monto) {
        if (monto <= 0) {
            throw new IllegalArgumentException("El monto a depositar debe ser positivo");
        }
        this.saldo += monto;
        return this.saldo;
    }

    /**
     * Retira un monto de la cuenta si hay saldo suficiente.
     *
     * @param  monto cantidad a retirar — debe ser positiva y ≤ saldo actual
     * @return {@code true} si el retiro fue exitoso, {@code false} si no hay saldo
     * @throws IllegalArgumentException si {@code monto} es menor o igual a cero
     */
    public boolean retirar(double monto) {
        if (monto <= 0) {
            throw new IllegalArgumentException("El monto a retirar debe ser positivo");
        }
        if (monto > this.saldo) {
            return false;   // saldo insuficiente
        }
        this.saldo -= monto;
        return true;
    }

    /**
     * Retorna una representación textual de la cuenta.
     * Equivalente al {@code __str__} de Python.
     *
     * @return String con titular y saldo formateado
     */
    @Override
    public String toString() {
        return String.format("Cuenta[%s | Saldo: $%.2f]", titular, saldo);
    }
}

### Convenciones de Codificación (Coding Standards)

#### 🔄 Contraste de Sintaxis

Python tiene **PEP 8**. Java tiene las **Oracle Java Code Conventions** y el estilo **Google Java Style Guide**. Las diferencias más importantes para un pythonista:

|Concepto|Python (PEP 8)|Java (Oracle/Google)|
|---|---|---|
|Nombres de clases|`MiClase` (CapWords)|`MiClase` (UpperCamelCase) — igual|
|Nombres de métodos|`mi_metodo()` (snake_case)|`miMetodo()` (lowerCamelCase)|
|Nombres de variables|`mi_variable` (snake_case)|`miVariable` (lowerCamelCase)|
|Constantes|`MI_CONSTANTE` (UPPER_SNAKE)|`MI_CONSTANTE` (UPPER_SNAKE) — igual|
|Paquetes|`mi_modulo`|`com.empresa.modulo` (todo minúsculas)|
|Longitud de línea|79 chars (PEP 8)|100 chars (Google)|
|Indentación|4 espacios|4 espacios — igual|
|Llaves de apertura|No aplica|Misma línea (`{` al final, no en línea nueva)|

// ✅ CONVENCIONES JAVA CORRECTAS — referencia rápida

// Paquete: todo en minúsculas, estilo dominio invertido
package com.miempresa.proyecto;

// Clase: UpperCamelCase — igual que Python
public class GestorDeUsuarios {

    // Constante de clase: UPPER_SNAKE_CASE — igual que Python
    private static final int MAX_INTENTOS = 3;

    // Atributo de instancia: lowerCamelCase — DIFERENTE a Python (snake_case)
    private String nombreUsuario;
    private int    intentosFallidos;

    // Constructor: mismo nombre que la clase
    public GestorDeUsuarios(String nombreUsuario) {
        this.nombreUsuario    = nombreUsuario;  // 'this.' distingue atributo de parámetro
        this.intentosFallidos = 0;
    }

    // Método: lowerCamelCase — DIFERENTE a Python (snake_case)
    public boolean verificarAcceso(String contrasena) {  // lowerCamelCase
        if (intentosFallidos >= MAX_INTENTOS) {
            return false;
        }
        // lógica de verificación...
        return true;
    }

    // Getter: getAtributo() — convención JavaBeans
    public String getNombreUsuario() {       // get + UpperCamelCase del atributo
        return nombreUsuario;
    }

    // Setter: setAtributo(valor) — convención JavaBeans
    public void setNombreUsuario(String nombreUsuario) {
        this.nombreUsuario = nombreUsuario;  // 'this.' necesario cuando coinciden nombres
    }
}
