## 👪 Herencia — `extends` y `super`

### 🔄 Contraste de Sintaxis

```
Python:                         Java:
────────────────────────        ──────────────────────────────
class Empleado(Persona):        public class Empleado extends Persona {
    def __init__(self, ...):
        super().__init__(...)       public Empleado(...) {
                                       super(...);   // DEBE ser la primera línea
                                   }
                                }
```

**Diferencias críticas:**

|Aspecto|Python|Java|
|---|---|---|
|Sintaxis|`class Hijo(Padre)`|`class Hijo extends Padre`|
|Herencia múltiple|✅ Directa|❌ Solo una clase (`extends`), múltiples interfaces (`implements`)|
|Llamar al padre|`super().__init__()`|`super(args)` — primera línea del constructor|
|Sobreescribir|Se hace automáticamente|Requiere `@Override` (convención/verificación)|
|Bloquear herencia|No estándar|`final class` — no puede extenderse|

### 💻 Bloque de Código — Jerarquía de herencia completa

```java
// Jerarquía: Persona → Empleado → Gerente

// ── CLASE BASE ─────────────────────────────────────────────────────────────
class Persona {

    private String nombre;
    private int    edad;

    public Persona(String nombre, int edad) {
        this.nombre = nombre;
        this.edad   = edad;
    }

    // Getters (protected permite acceso desde subclases del mismo paquete)
    public String getNombre() { return nombre; }
    public int    getEdad()   { return edad;   }

    public String presentarse() {
        return String.format("Soy %s, tengo %d años", nombre, edad);
    }

    @Override
    public String toString() {
        return "Persona{nombre='" + nombre + "', edad=" + edad + "}";
    }
}

// ── CLASE HIJA: extiende Persona ───────────────────────────────────────────
class Empleado extends Persona {

    private String cargo;
    private double salario;

    public Empleado(String nombre, int edad, String cargo, double salario) {
        // super() llama al constructor de Persona — OBLIGATORIO PRIMERA LÍNEA
        super(nombre, edad);

        // Luego inicializamos los atributos propios de Empleado
        this.cargo   = cargo;
        this.salario = salario;
    }

    public String getCargo()   { return cargo;   }
    public double getSalario() { return salario;  }

    public void aplicarAumento(double porcentaje) {
        this.salario *= (1 + porcentaje / 100.0);
    }

    // @Override: indica al compilador que estamos sobreescribiendo un método
    // Si el nombre está mal escrito, el compilador lo detecta — protección extra
    @Override
    public String presentarse() {
        // super.presentarse() reutiliza la lógica del padre
        return super.presentarse() + ", soy " + cargo;
    }

    @Override
    public String toString() {
        return String.format("Empleado{%s, cargo='%s', salario=$%.2f}",
                             getNombre(), cargo, salario);
    }
}

// ── CLASE NIETA: extiende Empleado ────────────────────────────────────────
class Gerente extends Empleado {

    private String departamento;
    private int    empleadosACargo;

    public Gerente(String nombre, int edad, double salario,
                   String departamento, int empleadosACargo) {
        super(nombre, edad, "Gerente", salario);   // llama al constructor de Empleado
        this.departamento      = departamento;
        this.empleadosACargo   = empleadosACargo;
    }

    // Método exclusivo de Gerente
    public void dirigirReunion() {
        System.out.printf("[%s] Convocando reunión del depto. %s (%d asistentes)%n",
                          getNombre(), departamento, empleadosACargo);
    }

    @Override
    public String presentarse() {
        // Cadena de super: llama a Empleado.presentarse() que llama a Persona.presentarse()
        return super.presentarse() + " del departamento de " + departamento;
    }

    @Override
    public String toString() {
        return String.format("Gerente{%s, depto='%s', equipo=%d}",
                             getNombre(), departamento, empleadosACargo);
    }
}

// ── PROGRAMA PRINCIPAL ─────────────────────────────────────────────────────
class DemoHerencia {
    public static void main(String[] args) {

        Persona  p = new Persona("Carlos", 45);
        Empleado e = new Empleado("Ana", 30, "Desarrolladora", 60000.0);
        Gerente  g = new Gerente("Luis", 40, 90000.0, "Tecnología", 8);

        System.out.println(p.presentarse());
        System.out.println(e.presentarse());  // encadena super
        System.out.println(g.presentarse());  // encadena dos super

        g.aplicarAumento(15);   // heredado de Empleado
        System.out.printf("Salario tras aumento: $%.2f%n", g.getSalario());

        g.dirigirReunion();

        // instanceof: verifica la cadena de herencia completa
        System.out.println(g instanceof Gerente);   // true
        System.out.println(g instanceof Empleado);  // true — Gerente ES Empleado
        System.out.println(g instanceof Persona);   // true — Gerente ES Persona
        System.out.println(e instanceof Gerente);   // false — Empleado NO ES Gerente

        // Polimorfismo: referencia de tipo Persona apunta a un Empleado
        Persona ref = new Empleado("Mia", 25, "Diseñadora", 50000.0);
        System.out.println(ref.presentarse());  // ejecuta Empleado.presentarse()
    }
}
```

### 🪲 "Rompe el Código"

```java
// ❌ ERROR 1: super() no es la primera línea del constructor

class Animal {
    String especie;
    Animal(String especie) { this.especie = especie; }
}

class Perro extends Animal {
    String nombre;

    Perro(String nombre) {
        this.nombre = nombre;   // ❌ error: call to super must be first statement
        super("Canis lupus");   // super() DEBE ir antes de cualquier otra sentencia
    }

    // ✅ SOLUCIÓN:
    Perro(String nombreCorregido) {
        super("Canis lupus");         // ✅ primera línea siempre
        this.nombre = nombreCorregido;
    }
}


// ❌ ERROR 2: @Override con nombre de método mal escrito (el compilador salva)

class Base {
    public void calcular() { System.out.println("Base"); }
}

class Hija extends Base {
    // Sin @Override, Java crea un NUEVO método en lugar de sobreescribir:
    public void calclar() {   // typo — diferente nombre
        System.out.println("Hija");
    }
    // calcular() de Base sigue sin sobreescribirse — bug silencioso

    // Con @Override, el compilador detecta que no existe 'calclar' en Base:
    @Override
    public void calclar() {   // ❌ error: method does not override or implement
    }                         //    a method from a supertype
    // ✅ El error de compilación te obliga a corregir el typo → calcular()
}
```
