#### 🔄 Contraste de Sintaxis

|Loop|Python|Java|Diferencia clave|
|---|---|---|---|
|`for` clásico|`for i in range(n)`|`for (int i=0; i<n; i++)`|Java expone init/condición/paso|
|`for-each`|`for item in lista`|`for (Tipo item : coleccion)`|Dos puntos `:` en Java|
|`while`|`while condicion:`|`while (condicion) { }`|Paréntesis y llaves|
|`do-while`|No existe|`do { } while (condicion);`|Ejecuta al menos una vez|

**El `for` clásico de Java** desglosa explícitamente lo que `range()` hace en Python:

```
Python:  for i in range(inicio, fin, paso):
Java:    for (int i = inicio; i < fin; i += paso)
              ↑                ↑           ↑
         inicialización   condición    actualización
```

#### 💻 Bloque de Código — Todos los tipos de loop

java

```java
// LoopsDemo.java

import java.util.ArrayList;
import java.util.List;

public class LoopsDemo {

    public static void main(String[] args) {

        // ── FOR CLÁSICO: control total sobre inicio, condición y paso ──────
        // Python: for i in range(0, 10, 2)
        for (int i = 0; i < 10; i += 2) {
            System.out.print(i + " ");  // 0 2 4 6 8
        }
        System.out.println();

        // Iteración inversa — Python: for i in range(9, -1, -1)
        for (int i = 9; i >= 0; i--) {
            System.out.print(i + " ");  // 9 8 7 6 5 4 3 2 1 0
        }
        System.out.println();

        // Múltiples variables en el for (no tiene equivalente directo en Python)
        for (int i = 0, j = 10; i < j; i++, j--) {
            System.out.print("(" + i + "," + j + ") ");  // (0,10) (1,9) (2,8)...
        }
        System.out.println();

        // ── FOR-EACH: iterar sobre arrays y colecciones ────────────────────
        // Python: for fruta in frutas:
        // Java:   for (String fruta : frutas)  ← dos puntos, no 'in'
        String[] frutas = {"manzana", "pera", "uva", "kiwi"};

        for (String fruta : frutas) {
            System.out.println(fruta.toUpperCase());
        }

        // For-each con List
        List<Integer> numeros = new ArrayList<>();
        numeros.add(10); numeros.add(20); numeros.add(30);

        for (int num : numeros) {
            System.out.print(num + " ");  // 10 20 30
        }
        System.out.println();

        // ── WHILE: igual que Python en lógica ─────────────────────────────
        int intentos    = 0;
        int maxIntentos = 3;
        boolean autenticado = false;

        while (intentos < maxIntentos && !autenticado) {
            intentos++;
            System.out.println("Intento " + intentos);
            if (intentos == 2) autenticado = true;
        }
        System.out.println("Autenticado: " + autenticado);

        // ── DO-WHILE: no existe en Python ─────────────────────────────────
        // Ejecuta el cuerpo PRIMERO y LUEGO verifica — garantiza al menos una ejecución
        int intentoMenu = 0;
        int opcion;

        do {
            intentoMenu++;
            opcion = intentoMenu;  // simula lectura de input
            System.out.println("Menú mostrado (intento " + intentoMenu + ")");
        } while (opcion < 1 || opcion > 3);  // evalúa DESPUÉS del cuerpo

        // ── LOOPS ANIDADOS: matrices ───────────────────────────────────────
        for (int fila = 0; fila < 3; fila++) {
            for (int col = 0; col < 3; col++) {
                System.out.printf("[%d,%d] ", fila, col);
            }
            System.out.println();
        }

        // ── LOOP INFINITO controlado con break ─────────────────────────────
        // Python: while True: ... break
        // Java:   while (true) { ... break; }
        int contador = 0;
        while (true) {
            contador++;
            if (contador >= 5) break;
        }
        System.out.println("Contador final: " + contador);  // 5
    }
}
```

```java
// ❌ ERROR 1: modificar variable del for-each (no afecta el array original)

int[] valores = {1, 2, 3, 4, 5};

for (int v : valores) {
    v = v * 2;  // ❌ modifica la copia LOCAL — el array original no cambia
}
System.out.println(valores[0]);  // sigue siendo 1, no 2

// ✅ SOLUCIÓN: usar for clásico con índice para modificar el array
for (int i = 0; i < valores.length; i++) {
    valores[i] = valores[i] * 2;  // ✅ modifica el elemento real
}


// ❌ ERROR 2: off-by-one con <= en lugar de <
int[] datos = {10, 20, 30};

for (int i = 0; i <= datos.length; i++) {   // ❌ <= permite i=3
    System.out.println(datos[i]);
}
// ArrayIndexOutOfBoundsException: Index 3 out of bounds for length 3

// ✅ SOLUCIÓN: siempre i < array.length
for (int i = 0; i < datos.length; i++) {    // ✅
    System.out.println(datos[i]);
}


// ❌ ERROR 3: do-while sin ; al final
do {
    System.out.println("Hola");
} while (false)    // ❌ falta ; — error: ';' expected

// ✅ El do-while es el ÚNICO loop que termina con ;
do {
    System.out.println("Hola");
} while (false);   // ✅
```
