### 6.1 Operaciones CRUD Avanzadas y Métodos Clave

  

```python

# ── Creación ───────────────────────────────────────────────────────────────

d = {"nombre": "Ana", "edad": 30, "activo": True}

d2 = dict(nombre="Luis", edad=25)       # usando keywords

d3 = dict(zip(["a", "b"], [1, 2]))      # desde dos listas

  

# ── READ: acceso y consulta ───────────────────────────────────────────────

print(d["nombre"])              # "Ana" — KeyError si no existe

print(d.get("nombre"))          # "Ana" — None si no existe (seguro)

print(d.get("ciudad", "N/A"))   # "N/A" — valor por defecto si no existe

  

# keys(), values(), items() retornan VISTAS (dinámicas, no copias)

for clave, valor in d.items():

    print(f"{clave}: {valor}")

  

# ── CREATE / UPDATE ───────────────────────────────────────────────────────

d["ciudad"] = "Madrid"          # agregar o sobreescribir

d.update({"edad": 31, "pais": "España"})   # múltiples a la vez

d.update(activo=False)          # con keywords también

  

# setdefault: asigna SOLO si la clave NO existe, y retorna el valor

# Patrón clásico para agrupar valores en listas

grupos = {}

datos = [("a", 1), ("b", 2), ("a", 3)]

for clave, valor in datos:

    grupos.setdefault(clave, []).append(valor)

print(grupos)   # {'a': [1, 3], 'b': [2]}

  

# ── DELETE ────────────────────────────────────────────────────────────────

valor = d.pop("ciudad")             # elimina y retorna el valor

valor = d.pop("inexistente", None)  # con default — no lanza KeyError

del d["activo"]                     # elimina, sin retornar

d.clear()                           # vacía el dict

  

# ── Merge de diccionarios (Python 3.9+) ──────────────────────────────────

base    = {"a": 1, "b": 2}

extra   = {"b": 99, "c": 3}

merged  = base | extra          # nuevo dict, extra tiene precedencia

base   |= extra                 # merge in-place

print(merged)   # {'a': 1, 'b': 99, 'c': 3}

```

### 6.2 Dict Comprehensions

  

```python

# ── Sintaxis: {clave: valor for item in iterable if condición} ─────────────

cuadrados = {x: x**2 for x in range(6)}

# {0:0, 1:1, 2:4, 3:9, 4:16, 5:25}

  

# ── Invertir clave-valor ──────────────────────────────────────────────────

original  = {"a": 1, "b": 2, "c": 3}

invertido = {v: k for k, v in original.items()}

# {1: 'a', 2: 'b', 3: 'c'}

  

# ── Con condición ─────────────────────────────────────────────────────────

empleados = {"Ana": 70000, "Luis": 35000, "Mia": 80000}

senior     = {nombre: salario

              for nombre, salario in empleados.items()

              if salario >= 50000}

# {'Ana': 70000, 'Mia': 80000}

  

# ── Caso real: frecuencia de caracteres ──────────────────────────────────

texto = "mississippi"

freq  = {c: texto.count(c) for c in set(texto)}

print(freq)   # {'m':1, 'i':4, 's':4, 'p':2}

```
### 6.3 `defaultdict` y `Counter`


```python

from collections import defaultdict, Counter

  

# ── defaultdict: como dict, pero genera un valor por defecto automáticamente

# Elimina la necesidad de setdefault() o try/except para claves nuevas

  

# Con list: agrupa elementos

grupos = defaultdict(list)

for nombre, dept in [("Ana","IT"), ("Luis","Ventas"), ("Mia","IT")]:

    grupos[nombre].append(dept)   # no hay KeyError la primera vez

print(dict(grupos))   # {'Ana': ['IT'], 'Luis': ['Ventas'], 'Mia': ['IT']}

  

# Con int (valor default = 0): conteo

conteo = defaultdict(int)

for letra in "mississippi":

    conteo[letra] += 1

print(dict(conteo))   # {'m':1, 'i':4, 's':4, 'p':2}

  

# Con set: grupos sin duplicados

grafo = defaultdict(set)

grafo["A"].add("B")

grafo["A"].add("C")

  

# ── Counter: subclase de dict para conteo, con superpoderes ──────────────

texto = "la casa es la base de la familia"

c = Counter(texto.split())

print(c)

# Counter({'la': 3, 'casa': 1, 'es': 1, 'base': 1, 'de': 1, 'familia': 1})

  

# Métodos especiales de Counter

print(c.most_common(2))    # [('la', 3), ('casa', 1)]

print(c.most_common()[:-3:-1])  # los 2 MENOS comunes (truco de slicing)

  

# Aritmética de Counters

c1 = Counter("aab")

c2 = Counter("abc")

print(c1 + c2)   # Counter({'a': 3, 'b': 2, 'c': 1})

print(c1 - c2)   # Counter({'a': 1})   — solo positivos

print(c1 & c2)   # Counter({'a': 1, 'b': 1})  — mínimo

print(c1 | c2)   # Counter({'a': 2, 'b': 1, 'c': 1})  — máximo

```
