Un **generador** es una función que produce valores de uno en uno, **pausando su ejecución** entre cada uno. En lugar de cargar todos los valores en memoria (como una lista), los genera bajo demanda (**lazy evaluation**).

  

```python

# ── Comparación directa ───────────────────────────────────────────────────

def cuadrados_lista(n):

    """Crea y retorna TODA la lista en memoria"""

    resultado = []

    for i in range(n):

        resultado.append(i ** 2)

    return resultado   # retorna todo de una vez

  

def cuadrados_gen(n):

    """Genera un valor a la vez, pausando con yield"""

    for i in range(n):

        yield i ** 2   # pausa aquí, devuelve i**2, y reanuda en la siguiente llamada

  

# ── Diferencia de uso ─────────────────────────────────────────────────────

lista = cuadrados_lista(5)   # [0, 1, 4, 9, 16] — TODO en memoria

gen   = cuadrados_gen(5)     # <generator object> — NADA en memoria aún

  

# El generador produce valores solo cuando se lo pedimos

print(next(gen))   # 0

print(next(gen))   # 1

# O iterar con for (lo más común)

for val in cuadrados_gen(5):

    print(val)

  

# ── Ventaja de memoria (caso extremo) ─────────────────────────────────────

import sys

  

lista_mem = [x**2 for x in range(1_000_000)]

gen_mem   = (x**2 for x in range(1_000_000))

  

print(sys.getsizeof(lista_mem))   # ~8 MB

print(sys.getsizeof(gen_mem))     # ~112 bytes ← sin importar el tamaño

  

# ── Generador con lógica compleja ─────────────────────────────────────────

def leer_bloques(ruta, tamano=4096):

    """Generador para leer archivos grandes sin cargarlos en memoria"""

    with open(ruta, 'rb') as f:

        while bloque := f.read(tamano):

            yield bloque
```
