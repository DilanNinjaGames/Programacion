### 8.1 Clases, `__init__`, Atributos de Instancia vs Atributos de Clase

  

```python

class Vehiculo:

    # ── Atributo de CLASE: compartido por TODAS las instancias ───────────

    total_vehiculos = 0   # contador global

  

    def __init__(self, marca, modelo, anio):

        # ── Atributos de INSTANCIA: únicos para cada objeto ──────────────

        self.marca  = marca

        self.modelo = modelo

        self.anio   = anio

        self._velocidad = 0        # convención _ = "privado" (no forzado)

        Vehiculo.total_vehiculos += 1

  

    def acelerar(self, kmh):

        """Método de instancia: opera sobre self"""

        self._velocidad += kmh

        return self   # permite method chaining: v.acelerar(50).acelerar(30)

  

    def estado(self):

        return f"{self.marca} {self.modelo} a {self._velocidad} km/h"

  
  

# ── Diferencia crítica: instancia vs clase ────────────────────────────────

v1 = Vehiculo("Toyota", "Corolla", 2022)

v2 = Vehiculo("Honda",  "Civic",   2023)

  

print(Vehiculo.total_vehiculos)   # 2  ← acceso por la CLASE

print(v1.total_vehiculos)         # 2  ← también accesible por instancia

  

v1.marca = "Ford"                  # modifica solo la instancia v1

print(v1.marca)                    # "Ford"

print(v2.marca)                    # "Honda" — no afectado

  

# ⚠️ Trampa: asignar atributo de clase desde una instancia crea un

# atributo de INSTANCIA que OCULTA al de clase (no lo modifica)

v1.total_vehiculos = 99            # crea v1.total_vehiculos (instancia)

print(Vehiculo.total_vehiculos)    # 2 — el atributo de clase no cambió

```

### 8.2 Herencia Simple y `super()`

  

```python

class Vehiculo:

    def __init__(self, marca, modelo):

        self.marca  = marca

        self.modelo = modelo

  

    def describir(self):

        return f"Vehículo: {self.marca} {self.modelo}"

  
  

class Auto(Vehiculo):

    """Auto hereda de Vehiculo"""

  

    def __init__(self, marca, modelo, num_puertas):

        # super() llama al __init__ del padre — siempre debe ir PRIMERO

        super().__init__(marca, modelo)

        # Luego agregamos los atributos propios de Auto

        self.num_puertas = num_puertas

  

    def describir(self):

        # Extendemos el método del padre, reutilizando su lógica

        base = super().describir()

        return f"{base} ({self.num_puertas} puertas)"

  
  

class AutoElectrico(Auto):

    """AutoElectrico hereda de Auto (herencia multinivel)"""

  

    def __init__(self, marca, modelo, num_puertas, autonomia_km):

        super().__init__(marca, modelo, num_puertas)

        self.autonomia_km = autonomia_km

  

    def describir(self):

        base = super().describir()

        return f"{base} — Autonomía: {self.autonomia_km} km"

  
  

# ── MRO (Method Resolution Order) ────────────────────────────────────────

# Python sigue el MRO (algoritmo C3) para resolver qué método usar

print(AutoElectrico.__mro__)

# (<class AutoElectrico>, <class Auto>, <class Vehiculo>, <class object>)

  

ae = AutoElectrico("Tesla", "Model 3", 4, 580)

print(ae.describir())

# "Vehículo: Tesla Model 3 (4 puertas) — Autonomía: 580 km"

  

print(isinstance(ae, Auto))      # True — ae es instancia de Auto

print(isinstance(ae, Vehiculo))  # True — y de Vehiculo

print(issubclass(Auto, Vehiculo)) # True

```

### 8.3 Polimorfismo y Duck Typing

  

```python

# ── Polimorfismo: el mismo método actúa diferente según el objeto ──────────

class Perro:

    def sonido(self):

        return "Guau"

  

class Gato:

    def sonido(self):

        return "Miau"

  

class Pato:

    def sonido(self):

        return "Cuac"

  

animales = [Perro(), Gato(), Pato()]

for animal in animales:

    print(animal.sonido())   # Guau / Miau / Cuac — sin isinstance ni if

  

# ── Duck Typing: "Si camina como pato y grazna como pato, es un pato" ─────

# Python no exige herencia formal — solo que el objeto tenga el método

  

class Robot:

    """No hereda de ningún Animal, pero tiene el método sonido()"""

    def sonido(self):

        return "Beep boop"

  

animales.append(Robot())    # funciona igual en el bucle anterior

  

# ── EAFP vs LBYL ──────────────────────────────────────────────────────────

# LBYL (Look Before You Leap) — estilo C/Java: verificar antes

def procesar_lbyl(obj):

    if hasattr(obj, 'sonido'):

        obj.sonido()

  

# EAFP (Easier to Ask Forgiveness than Permission) — estilo Python preferido

def procesar_eafp(obj):

    try:

        obj.sonido()

    except AttributeError:

        print("El objeto no tiene método sonido()")

  

# ── Protocolo formal con ABC (Abstract Base Class) ────────────────────────

from abc import ABC, abstractmethod

  

class Animal(ABC):

    @abstractmethod

    def sonido(self):

        """Las subclases DEBEN implementar este método"""

        pass

  

    def dormir(self):

        return "Zzz..."  # método concreto — puede heredarse tal cual

  

class Leon(Animal):

    def sonido(self):   # obligatorio implementar

        return "Roar"

  

# Animal()  # ❌ TypeError: no se puede instanciar clase abstracta

l = Leon()

print(l.sonido())   # "Roar"

print(l.dormir())   # "Zzz..."

```

### 8.4 Métodos Especiales (Dunder Methods)

  

Los **dunder methods** (`__method__`) permiten que tus objetos se integren con la sintaxis de Python (operadores, `len()`, `print()`, comparaciones, etc.).

  

```python

class Vector:

    def __init__(self, x, y):

        self.x = x

        self.y = y

  

    # ── Representación ────────────────────────────────────────────────────

    def __repr__(self):

        """Para desarrolladores: debe ser unambiguo y reproducible"""

        return f"Vector({self.x}, {self.y})"

  

    def __str__(self):

        """Para usuarios: legible y amigable"""

        return f"({self.x}, {self.y})"

  

    # ── Longitud ──────────────────────────────────────────────────────────

    def __len__(self):

        """Permite usar len(vector)"""

        return int((self.x**2 + self.y**2) ** 0.5)

  

    # ── Comparación ───────────────────────────────────────────────────────

    def __eq__(self, other):

        """Permite usar == entre vectores"""

        if not isinstance(other, Vector):

            return NotImplemented   # no lanzar TypeError, retornar NotImplemented

        return self.x == other.x and self.y == other.y

  

    def __lt__(self, other):

        """Permite < y habilita sorted() con objetos Vector"""

        return len(self) < len(other)

  

    # ── Operadores aritméticos ────────────────────────────────────────────

    def __add__(self, other):

        """Permite v1 + v2"""

        return Vector(self.x + other.x, self.y + other.y)

  

    def __mul__(self, escalar):

        """Permite v * 3"""

        return Vector(self.x * escalar, self.y * escalar)

  

    def __rmul__(self, escalar):

        """Permite 3 * v (cuando el escalar está a la izquierda)"""

        return self.__mul__(escalar)

  

    # ── Iterabilidad ──────────────────────────────────────────────────────

    def __iter__(self):

        """Permite: x, y = vector  y  for c in vector"""

        yield self.x

        yield self.y

  

    # ── Contexto (with statement) ─────────────────────────────────────────

    def __enter__(self):

        print("Entrando al contexto")

        return self

  

    def __exit__(self, exc_type, exc_val, exc_tb):

        print("Saliendo del contexto")

        return False   # False = no suprimir excepciones

  
  

# ── Prueba ─────────────────────────────────────────────────────────────────

v1 = Vector(3, 4)

v2 = Vector(1, 2)

  

print(repr(v1))     # Vector(3, 4)   — usado en REPL y debugging

print(str(v1))      # (3, 4)         — usado en print()

print(len(v1))      # 5              — magnitud entera

print(v1 == v2)     # False

print(v1 + v2)      # (4, 6)

print(v1 * 3)       # (9, 12)

print(3 * v1)       # (9, 12)

  

x, y = v1           # unpacking gracias a __iter__

print(x, y)         # 3  4

  

with Vector(0, 0):  # context manager gracias a __enter__/__exit__

    print("Dentro del with")

```

### 8.5 Decoradores Nativos: `@property`, `@classmethod`, `@staticmethod`

  

```python

class Temperatura:

    """Ejemplo canónico de los tres decoradores nativos"""

  

    # ── Atributo de clase ─────────────────────────────────────────────────

    unidad_default = "Celsius"

  

    def __init__(self, celsius):

        self._celsius = celsius   # _ indica "privado por convención"

  

    # ── @property: getter ─────────────────────────────────────────────────

    @property

    def celsius(self):

        """Acceso controlado: t.celsius sin paréntesis"""

        return self._celsius

  

    # ── @<propiedad>.setter ───────────────────────────────────────────────

    @celsius.setter

    def celsius(self, valor):

        """Validación automática al asignar: t.celsius = -300"""

        if valor < -273.15:

            raise ValueError("Temperatura bajo el cero absoluto")

        self._celsius = valor

  

    # ── @<propiedad>.deleter ──────────────────────────────────────────────

    @celsius.deleter

    def celsius(self):

        print("Eliminando temperatura")

        del self._celsius

  

    # ── @property calculada (solo getter, no setter) ──────────────────────

    @property

    def fahrenheit(self):

        """Calculada en tiempo real, nunca almacenada"""

        return self._celsius * 9/5 + 32

  

    @property

    def kelvin(self):

        return self._celsius + 273.15

  

    # ── @classmethod: recibe la CLASE (cls) en lugar de la instancia ──────

    # Usado para: fábricas alternativas, métodos que operan sobre la clase

    @classmethod

    def desde_fahrenheit(cls, fahrenheit):

        """Factory method: crea instancia desde Fahrenheit"""

        celsius = (fahrenheit - 32) * 5/9

        return cls(celsius)   # cls() es como Temperatura()

  

    @classmethod

    def cambiar_unidad_default(cls, unidad):

        cls.unidad_default = unidad   # modifica el atributo de clase

  

    # ── @staticmethod: función de utilidad relacionada, sin acceso a cls/self

    # No recibe self ni cls — es simplemente una función en el namespace de la clase

    @staticmethod

    def es_valida(valor):

        """Validación de utilidad — no necesita acceso al estado"""

        return valor >= -273.15

  
  

# ── Uso ────────────────────────────────────────────────────────────────────

t = Temperatura(100)

  

# @property: se accede como atributo (sin paréntesis)

print(t.celsius)        # 100

print(t.fahrenheit)     # 212.0

print(t.kelvin)         # 373.15

  

# @setter: se asigna como atributo

t.celsius = 37

print(t.fahrenheit)     # 98.6

  

# @setter con validación

try:

    t.celsius = -300    # ❌ ValueError

except ValueError as e:

    print(e)

  

# @classmethod como factory

t2 = Temperatura.desde_fahrenheit(32)

print(t2.celsius)       # 0.0

  

# @staticmethod: llamado en la clase o instancia (sin estado)

print(Temperatura.es_valida(-300))    # False

print(Temperatura.es_valida(20))      # True

```
