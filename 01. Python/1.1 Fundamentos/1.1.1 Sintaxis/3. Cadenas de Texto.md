### 3.1 Creación, Indexación y Slicing

  

```python

# ── Creación ───────────────────────────────────────────────────────────────

s1 = 'comillas simples'

s2 = "comillas dobles"

s3 = """cadena

multilínea"""

s4 = r"C:\nueva\ruta"      # raw string: \ no es secuencia de escape

s5 = b"bytes"              # bytes literal (no es str)

  

texto = "Python"

#        0123456  ← índices positivos

#       -6-5-4-3-2-1  ← índices negativos

  

# ── Indexación ────────────────────────────────────────────────────────────

print(texto[0])    # 'P'

print(texto[-1])   # 'n'   ← último carácter

  

# ── Slicing: s[inicio:fin_exclusivo:paso] ─────────────────────────────────

print(texto[1:4])   # 'yth'

print(texto[:3])    # 'Pyt'  — desde el principio

print(texto[3:])    # 'hon'  — hasta el final

print(texto[::-1])  # 'nohtyP'  — inversión

print(texto[::2])   # 'Pto'  — un carácter de cada dos

```

### 3.2 Métodos Nativos Clave

  

```python

s = "  Hola, Mundo Python  "

  

# ── Limpieza ──────────────────────────────────────────────────────────────

print(s.strip())          # "Hola, Mundo Python"  — quita espacios extremos

print(s.lstrip())         # "Hola, Mundo Python  "

print(s.rstrip())         # "  Hola, Mundo Python"

  

# ── Búsqueda ──────────────────────────────────────────────────────────────

print(s.find("Mundo"))    # 7   — índice o -1 si no existe

print(s.index("Mundo"))   # 7   — igual pero lanza ValueError si no existe

print("Python" in s)      # True — preferible para booleano

  

# ── Transformación ────────────────────────────────────────────────────────

print(s.strip().upper())          # "HOLA, MUNDO PYTHON"

print(s.strip().lower())          # "hola, mundo python"

print(s.strip().title())          # "Hola, Mundo Python"

print(s.strip().replace(",", "")) # "Hola Mundo Python"

  

# ── División y unión ──────────────────────────────────────────────────────

csv_line = "ana,30,developer"

partes = csv_line.split(",")        # ['ana', '30', 'developer']

print(partes)

  

# join() es el opuesto de split() — une listas de strings

print(" | ".join(partes))           # "ana | 30 | developer"

print("\n".join(partes))            # cada elemento en una línea

  

# ── Verificación de contenido ─────────────────────────────────────────────

print("42".isdigit())       # True

print("abc".isalpha())      # True

print("abc123".isalnum())   # True

print("  ".isspace())       # True

print("hello".startswith("he"))   # True

print("hello".endswith("lo"))     # True

  

# ── format() clásico (antes de f-strings) ────────────────────────────────

print("Hola, {}. Tienes {} años".format("Ana", 30))

print("Hola, {nombre}".format(nombre="Luis"))

```

### 3.3 f-strings Avanzados

  

Los f-strings (Python 3.6+) son la forma **más eficiente y legible** de formatear cadenas.

  

```python

nombre = "Ana"

pi     = 3.14159265

precio = 1234567.89

nota   = 0.857

  

# ── Básico ────────────────────────────────────────────────────────────────

print(f"Hola, {nombre}!")               # "Hola, Ana!"

print(f"Pi = {pi}")                     # "Pi = 3.14159265"

  

# ── Formato numérico ──────────────────────────────────────────────────────

print(f"Pi redondeado:  {pi:.2f}")      # "3.14"

print(f"Notación cient: {pi:.2e}")      # "3.14e+00"

print(f"Porcentaje:     {nota:.1%}")    # "85.7%"

print(f"Con separador:  {precio:,.2f}") # "1,234,567.89"

print(f"Separador ES:   {precio:_.2f}") # "1_234_567.89"

  

# ── Relleno y alineación ──────────────────────────────────────────────────

# Sintaxis: {valor:[relleno][alineación][ancho]}

# Alineación: < (izq), > (der), ^ (centro)

print(f"{'izq':<10}|")   # "izq       |"

print(f"{'der':>10}|")   # "       der|"

print(f"{'cen':^10}|")   # "   cen    |"

  

# Con carácter de relleno personalizado

print(f"{'42':0>5}")     # "00042"  — relleno con ceros a la derecha

print(f"{'*':*^20}")     # "********** **********" ← no, rellena con *

  

# ── Expresiones dentro de f-strings ──────────────────────────────────────

x = 10

print(f"El doble es {x * 2}")          # "El doble es 20"

print(f"Lista: {[i**2 for i in range(4)]}")  # "Lista: [0, 1, 4, 9]"

  

# ── Debug con = (Python 3.8+) ─────────────────────────────────────────────

# Imprime nombre_variable=valor — extremadamente útil para debugging

print(f"{nombre=}")   # nombre='Ana'

print(f"{pi=:.4f}")   # pi=3.1416

  

# ── Formato de enteros ────────────────────────────────────────────────────

n = 255

print(f"Decimal: {n:d}")    # "255"

print(f"Binario: {n:b}")    # "11111111"

print(f"Hex:     {n:x}")    # "ff"

print(f"Hex may: {n:X}")    # "FF"

print(f"Octal:   {n:o}")    # "377"

```
