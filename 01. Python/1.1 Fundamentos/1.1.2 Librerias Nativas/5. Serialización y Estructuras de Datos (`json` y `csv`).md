### 5.1 Módulo `json`

  

**Teoría:** JSON (JavaScript Object Notation) es el formato de intercambio de datos dominante en APIs REST, configuraciones y pipelines de ML. Python mapea tipos JSON a tipos Python así:

  

| JSON | Python |

|---|---|

| `object {}` | `dict` |

| `array []` | `list` |

| `string ""` | `str` |

| `number` | `int` / `float` |

| `true/false` | `True/False` |

| `null` | `None` |

  

Las funciones siguen la convención: `s` al final = trabaja con **s**trings en memoria; sin `s` = trabaja con **f**ile objects.

  

```python

import json

from datetime import datetime

from pathlib import Path

  

datos = {

    "nombre": "Ana García",

    "edad": 30,

    "activo": True,

    "score": 9.5,

    "tags": ["python", "AI"],

    "direccion": None

}

  

# ── En memoria: dumps/loads ───────────────────────────────────────────────

# dumps: dict → string JSON

json_str = json.dumps(datos)

print(type(json_str))   # <class 'str'>

print(json_str)

# '{"nombre": "Ana García", "edad": 30, ...}'

  

# Con opciones de formato

json_bonito = json.dumps(datos, indent=4, ensure_ascii=False, sort_keys=True)

print(json_bonito)

# ensure_ascii=False: permite caracteres no-ASCII (acentos, ñ, etc.)

# sort_keys=True: claves ordenadas alfabéticamente (reproducible)

  

# loads: string JSON → dict

datos_recuperados = json.loads(json_str)

print(type(datos_recuperados))   # <class 'dict'>

print(datos_recuperados["nombre"])   # 'Ana García'

  

# ── Con archivos: dump/load ───────────────────────────────────────────────

ruta = Path('datos.json')

  

# Escribir

with open(ruta, 'w', encoding='utf-8') as f:

    json.dump(datos, f, indent=2, ensure_ascii=False)

  

# Leer

with open(ruta, 'r', encoding='utf-8') as f:

    datos_leidos = json.load(f)

  

# Con pathlib (más conciso para archivos pequeños)

ruta.write_text(json.dumps(datos, indent=2, ensure_ascii=False), encoding='utf-8')

datos_leidos = json.loads(ruta.read_text(encoding='utf-8'))

  

# ── Serialización de tipos no-JSON: encoder personalizado ────────────────

# json no sabe serializar datetime, Path, sets, etc.

# Solución: subclasificar JSONEncoder o usar el parámetro default=

  

class EncoderExtendido(json.JSONEncoder):

    def default(self, obj):

        if isinstance(obj, datetime):

            return obj.isoformat()        # datetime → string ISO 8601

        if isinstance(obj, Path):

            return str(obj)               # Path → string

        if isinstance(obj, set):

            return sorted(list(obj))      # set → list ordenada

        return super().default(obj)       # tipo no manejado → TypeError

  

evento = {

    "titulo": "Lanzamiento",

    "fecha": datetime(2024, 6, 15, 9, 0),

    "ruta": Path('/logs/evento.log'),

    "tags": {"python", "api"}

}

  

json_str = json.dumps(evento, cls=EncoderExtendido, indent=2)

print(json_str)

  

# Alternativa con parámetro default= (más simple para casos simples)

json.dumps(evento, default=str)   # convierte todo lo no-serializable con str()

  

# ── Deserialización: object_hook ──────────────────────────────────────────

def decodificar_fechas(d):

    """Convierte strings ISO 8601 de vuelta a datetime al cargar"""

    for k, v in d.items():

        if isinstance(v, str):

            try:

                d[k] = datetime.fromisoformat(v)

            except ValueError:

                pass

    return d

  

datos_ricos = json.loads(json_str, object_hook=decodificar_fechas)

```

### 5.2 Módulo `csv`

  

**Teoría:** CSV (Comma-Separated Values) carece de estándar formal, lo que genera variaciones: separador `;` en Excel europeo, campos con comas entre comillas, encoding variable. El módulo `csv` maneja estas ambigüedades mediante **dialectos** y respetando el estándar RFC 4180.

  

`csv.reader` retorna listas de strings (acceso por índice). `csv.DictReader` retorna dicts (acceso por nombre de columna) — más legible y robusto ante cambios en el orden de columnas.

  

```python

import csv

from pathlib import Path

  

# ── Escritura con csv.writer ──────────────────────────────────────────────

empleados = [

    ["nombre", "departamento", "salario"],   # cabecera

    ["Ana García",  "IT",      70000],

    ["Luis López",  "Ventas",  45000],

    ["Mia Chen",    "IT",      80000],

]

  

with open('empleados.csv', 'w', newline='', encoding='utf-8') as f:

    # newline='' es OBLIGATORIO en Windows para evitar líneas en blanco extra

    writer = csv.writer(f, delimiter=',', quoting=csv.QUOTE_MINIMAL)

    writer.writerows(empleados)   # escribe todas las filas de una vez

  

# ── Lectura con csv.reader ────────────────────────────────────────────────

with open('empleados.csv', 'r', encoding='utf-8') as f:

    reader = csv.reader(f)

    cabecera = next(reader)        # consumir/saltarse la primera fila

    print(f"Columnas: {cabecera}")

    for fila in reader:

        nombre, dept, salario = fila   # unpacking directo

        print(f"{nombre} ({dept}): ${int(salario):,}")

  

# ── Escritura con csv.DictWriter ──────────────────────────────────────────

datos = [

    {"nombre": "Ana",  "departamento": "IT",     "salario": 70000},

    {"nombre": "Luis", "departamento": "Ventas",  "salario": 45000},

]

campos = ["nombre", "departamento", "salario"]

  

with open('empleados_dict.csv', 'w', newline='', encoding='utf-8') as f:

    writer = csv.DictWriter(f, fieldnames=campos)

    writer.writeheader()       # escribe la fila de cabecera

    writer.writerows(datos)    # escribe todas las filas

  

# ── Lectura con csv.DictReader (preferido) ────────────────────────────────

with open('empleados_dict.csv', 'r', encoding='utf-8') as f:

    reader = csv.DictReader(f)   # usa la primera fila como claves

    empleados = list(reader)     # lista de OrderedDicts (o dicts en 3.8+)

  

# Ahora acceso por nombre de columna — robusto ante reordenamientos

for e in empleados:

    print(f"{e['nombre']}: {e['salario']}")

  

# ── Dialectos: manejar variaciones de CSV ────────────────────────────────

# Excel europeo usa ; como separador

with open('europeo.csv', 'r', encoding='utf-8') as f:

    reader = csv.reader(f, delimiter=';')

    for fila in reader:

        print(fila)

  

# Archivos TSV (Tab-Separated Values)

with open('datos.tsv', 'r') as f:

    reader = csv.reader(f, delimiter='\t')

    for fila in reader:

        print(fila)

  

# ── Caso real: filtrar y transformar CSV ──────────────────────────────────

with open('empleados.csv', 'r', encoding='utf-8') as entrada, \

     open('it_senior.csv', 'w', newline='', encoding='utf-8') as salida:

  

    reader = csv.DictReader(entrada)

    campos_salida = ['nombre', 'salario', 'categoria']

    writer = csv.DictWriter(salida, fieldnames=campos_salida)

    writer.writeheader()

  

    for fila in reader:

        if fila['departamento'] == 'IT' and int(fila['salario']) >= 60000:

            writer.writerow({

                'nombre':    fila['nombre'],

                'salario':   int(fila['salario']),

                'categoria': 'Senior'

            })

```
